# Anpassen /boot/config.txt   -----------------------------------------------

echo -e "\nmax_usb_current=1" >> /rootfs/boot/config.txt

# Installation von Dateien (mit neuem Installer automatisch)   --------------

  echo "Installiere Projektdateien ..."
  install_files dateien.list
fi
